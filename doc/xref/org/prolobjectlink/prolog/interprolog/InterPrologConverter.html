63  	public final PrologTerm toTerm(TermModel prologTerm) {
64  		if (prologTerm.isAtom()) {
65  			String functor = prologTerm.toString();
66  			if (functor.equals("nil")) {
67  				return new InterPrologNil(provider);
68  			} else if (functor.equals("!")) {
69  				return new InterPrologCut(createProvider());
70  			} else if (functor.equals("fail")) {
71  				return new InterPrologFail(provider);
72  			} else if (functor.equals("true")) {
73  				return new InterPrologTrue(provider);
74  			} else if (functor.equals("false")) {
75  				return new InterPrologFalse(provider);
76  			} else if (functor.equals("[]")) {
77  				return new InterPrologEmpty(provider);
78  			}
79  			return new InterPrologAtom(provider, functor);
80  		} else if (prologTerm.equals(InterPrologList.EMPTY)) {
81  			return new InterPrologEmpty(provider);
82  		} else if ((prologTerm.node instanceof Float) || (prologTerm.node instanceof BigDecimal)) {
83  			return new InterPrologFloat(provider, ((Number) prologTerm.node).floatValue());
84  		} else if ((prologTerm.node instanceof Double) || (prologTerm.node instanceof BigDecimal)) {
85  			return new InterPrologDouble(provider, ((Number) prologTerm.node).doubleValue());
86  		} else if (prologTerm.isInteger() || prologTerm.node instanceof BigInteger) {
87  			return new InterPrologInteger(provider, prologTerm.intValue());
88  		} else if (prologTerm.isLong() || prologTerm.node instanceof BigInteger) {
89  			return new InterPrologLong(provider, prologTerm.longValue());
90  		} else if (prologTerm.isVar()) {
91  			String name = ((TermVariable) prologTerm).getName();
92  			PrologVariable variable = sharedVariables.get(name);
93  			if (variable == null) {
94  				variable = new InterPrologVariable(provider, name);
95  				sharedVariables.put(variable.getName(), variable);
96  			}
97  			return variable;
98  		} else if (prologTerm.isList()) {
99  			return new InterPrologList(provider, prologTerm.flatList());
100 		} else if (prologTerm.getChildCount() > 0) {
101 
102 			TermModel compound = prologTerm;
103 			int arity = compound.getChildCount();
104 			String functor = compound.node.toString();
105 			TermModel[] arguments = new TermModel[arity];
106 
107 			if (arity == 2) {
108 				String key = "LIST";
109 				String stringQuery = "findall(P/S/O,current_op(P,S,O)," + key + "), buildTermModel(" + key + ",TM)";
110 
111 				// busy wait necessary to wait engine disposition
112 				if (!InterPrologEngine.engine.isIdle()) {
113 					InterPrologEngine.engine.waitUntilIdle();
114 				}
115 
116 				SolutionIterator si = InterPrologEngine.engine.goal(stringQuery, "[TM]");
117 				while (si.hasNext()) {
118 					Object[] bindings = si.next();
119 					for (Object object : bindings) {
120 						if (object instanceof TermModel) {
121 							TermModel list = (TermModel) object;
122 							while (list.getChildCount() > 0) {
123 								TermModel solvedTerm = (TermModel) list.getChild(0);
124 								String n = (String) solvedTerm.children[1].node;
125 								if (functor.contains("'" + n + "'")) {
126 									TermModel left = (TermModel) compound.getChild(0);
127 									TermModel right = (TermModel) compound.getChild(1);
128 									return new InterPrologStructure(provider, left, n, right);
129 								}
130 								list = (TermModel) list.getChild(1);
131 							}
132 						}
133 					}
134 				}
135 			}
136 
137 			for (int i = 0; i < arity; i++) {
138 				arguments[i] = (TermModel) compound.getChild(i);
139 			}
140 			return new InterPrologStructure(provider, functor, arguments);
141 
142 		}
143 
144 		throw new UnknownTermError(prologTerm);
145 	}

147 	public final TermModel fromTerm(PrologTerm term) {
148 		switch (term.getType()) {
149 		case NIL_TYPE:
150 			return new TermModel("nil");
151 		case CUT_TYPE:
152 			return new TermModel("!");
153 		case FAIL_TYPE:
154 			return new TermModel("fail");
155 		case TRUE_TYPE:
156 			return new TermModel("true");
157 		case FALSE_TYPE:
158 			return new TermModel("false");
159 		case ATOM_TYPE:
160 			return new TermModel(removeQuoted(((PrologAtom) term).getStringValue()));
161 		case FLOAT_TYPE:
162 			return new TermModel(((PrologFloat) term).getFloatValue());
163 		case INTEGER_TYPE:
164 			return new TermModel(((PrologInteger) term).getIntegerValue());
165 		case DOUBLE_TYPE:
166 			return new TermModel(((PrologDouble) term).getDoubleValue());
167 		case LONG_TYPE:
168 			return new TermModel(((PrologLong) term).getLongValue());
169 		case VARIABLE_TYPE:
170 			String name = ((PrologVariable) term).getName();
171 			int position = ((PrologVariable) term).getPosition();
172 			TermModel variable = sharedPrologVariables.get(name);
173 			if (variable == null) {
174 				variable = new TermVariable(name, position);
175 				sharedPrologVariables.put(name, variable);
176 			}
177 			return variable;
178 		case LIST_TYPE:
179 			PrologTerm[] array = term.getArguments();
180 			if (array.length < 1) {
181 				return InterPrologEmpty.EMPTY;
182 			}
183 			return TermModel.makeList(fromTermArray(array));
184 		case STRUCTURE_TYPE:
185 			String functor = term.getFunctor();
186 			TermModel[] arguments = fromTermArray(((PrologStructure) term).getArguments());
187 			if (functor.contains("op")) {
188 				TermModel x = parser.parseTerm(functor);
189 				TermModel ox = (TermModel) x.getChild(2);
190 				return new TermModel(ox.node, arguments);
191 			}
192 			return new TermModel(functor, arguments);
193 		default:
194 			throw new UnknownTermError(term);
195 		}
196 	}

198 	public final TermModel[] fromTermArray(PrologTerm[] terms) {
199 		TermModel[] prologTerms = new TermModel[terms.length];
200 		for (int i = 0; i < terms.length; i++) {
201 			prologTerms[i] = fromTerm(terms[i]);
202 		}
203 		return prologTerms;
204 	}
205 
206 	public final TermModel fromTerm(PrologTerm head, PrologTerm[] body) {
207 		TermModel clauseHead = fromTerm(head);
208 		if (body != null && body.length > 0) {
209 			TermModel clauseBody = fromTerm(body[body.length - 1]);
210 			for (int i = body.length - 2; i >= 0; --i) {
211 				clauseBody = new TermModel(",", new TermModel[] { fromTerm(body[i]), clauseBody });
212 			}
213 			return new TermModel(":-", new TermModel[] { clauseHead, clauseBody });
214 		}
215 		return clauseHead;
216 	}
